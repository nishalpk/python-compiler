  WHILE LPAREN Expr RPAREN LBRACE Expr ASSIGN Expr ADDOP Factor                              ; } $                                                                  REDUCE Term -> Factor                             
  WHILE LPAREN Expr RPAREN LBRACE Expr ASSIGN Expr ADDOP Term                                ; } $                                                                  REDUCE Expr -> Term

  WHILE LPAREN Expr RPAREN LBRACE Expr ASSIGN Expr SEMI                                      } $                                                                    REDUCE Assign -> Expr ASSIGN Expr SEMI

  WhileStmt                                                                                  $                                                                      REDUCE Stmt -> WhileStmt                          
  Stmt                                                                                       $                                                                      REDUCE StmtList -> Stmt
